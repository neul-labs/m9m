# n8n-go v0.2.0 Enhanced Platform Roadmap

**Timeline**: Q2 2024 (8-10 weeks from beta feedback)
**Focus**: Additional features, enhanced usability, and expanded node ecosystem

## Overview

Building on the successful v0.1.0 beta release, v0.2.0 will expand n8n-go's capabilities with additional node types, enhanced scheduling, RESTful API, and performance optimizations. This release focuses on making n8n-go a more complete platform while maintaining the 18x performance advantage.

## Goals for v0.2.0

### Primary Objectives
1. **Expand Node Ecosystem**: Add 8-10 new essential node types
2. **Enhanced Scheduling**: Built-in cron scheduler with web interface
3. **RESTful API**: Complete workflow management API
4. **Performance Optimization**: Further improvements and monitoring
5. **Developer Experience**: Better tooling and debugging capabilities

### Success Metrics
- **Node Coverage**: Support 95% of common n8n use cases
- **API Completeness**: Full CRUD operations for workflows
- **Performance**: Maintain 18x performance advantage
- **Developer Satisfaction**: Enhanced development experience

### File Operations Implementation

#### Read Binary File Node
```go
func (n *ReadBinaryFileNode) Execute(inputData []model.DataItem, nodeParams map[string]interface{}) ([]model.DataItem, error) {
    var results []model.DataItem

    for index, item := range inputData {
        // Create expression context
        context := &expressions.ExpressionContext{
            ActiveNodeName:      "Read Binary File",
            RunIndex:           0,
            ItemIndex:          index,
            Mode:               expressions.ModeManual,
            ConnectionInputData: []model.DataItem{item},
            AdditionalKeys:     make(map[string]interface{}),
        }

        // Get file path
        filePath, err := n.evaluator.EvaluateExpression(nodeParams["filePath"].(string), context)
        if err != nil {
            return nil, n.CreateError("failed to evaluate file path", map[string]interface{}{
                "expression": nodeParams["filePath"],
                "error":      err.Error(),
            })
        }

        // Validate file path
        if err := n.validateFilePath(filePath.(string)); err != nil {
            return nil, n.CreateError("invalid file path", map[string]interface{}{
                "path":  filePath,
                "error": err.Error(),
            })
        }

        // Read file
        data, err := os.ReadFile(filePath.(string))
        if err != nil {
            return nil, n.CreateError("failed to read file", map[string]interface{}{
                "path":  filePath,
                "error": err.Error(),
            })
        }

        // Process encoding if specified
        encoding := nodeParams["encoding"]
        var result interface{}
        switch encoding {
        case "base64":
            result = base64.StdEncoding.EncodeToString(data)
        case "hex":
            result = hex.EncodeToString(data)
        default:
            result = data
        }

        // Create result item
        resultItem := model.DataItem{
            JSON: map[string]interface{}{
                "data":     result,
                "filePath": filePath,
                "size":     len(data),
                "encoding": encoding,
            },
            Binary: data,
        }

        results = append(results, resultItem)
    }

    return results, nil
}

func (n *ReadBinaryFileNode) validateFilePath(path string) error {
    // Prevent directory traversal
    if strings.Contains(path, "..") {
        return errors.New("directory traversal not allowed")
    }

    // Check file exists
    if _, err := os.Stat(path); os.IsNotExist(err) {
        return errors.New("file does not exist")
    }

    return nil
}
```

### Email Integration Implementation

#### SMTP Send Email Node
```go
func (n *SMTPNode) Execute(inputData []model.DataItem, nodeParams map[string]interface{}) ([]model.DataItem, error) {
    var results []model.DataItem

    // Get SMTP configuration
    smtpConfig, err := n.getSMTPConfig(nodeParams)
    if err != nil {
        return nil, n.CreateError("invalid SMTP configuration", map[string]interface{}{
            "error": err.Error(),
        })
    }

    for index, item := range inputData {
        // Create expression context
        context := &expressions.ExpressionContext{
            ActiveNodeName:      "Send Email",
            RunIndex:           0,
            ItemIndex:          index,
            Mode:               expressions.ModeManual,
            ConnectionInputData: []model.DataItem{item},
            AdditionalKeys:     make(map[string]interface{}),
        }

        // Evaluate email parameters
        to, err := n.evaluator.EvaluateExpression(nodeParams["to"].(string), context)
        if err != nil {
            return nil, n.CreateError("failed to evaluate 'to' field", map[string]interface{}{
                "expression": nodeParams["to"],
                "error":      err.Error(),
            })
        }

        subject, err := n.evaluator.EvaluateExpression(nodeParams["subject"].(string), context)
        if err != nil {
            return nil, n.CreateError("failed to evaluate subject", map[string]interface{}{
                "expression": nodeParams["subject"],
                "error":      err.Error(),
            })
        }

        body, err := n.evaluator.EvaluateExpression(nodeParams["body"].(string), context)
        if err != nil {
            return nil, n.CreateError("failed to evaluate body", map[string]interface{}{
                "expression": nodeParams["body"],
                "error":      err.Error(),
            })
        }

        // Send email
        messageID, err := n.sendEmail(smtpConfig, EmailMessage{
            To:      to.(string),
            Subject: subject.(string),
            Body:    body.(string),
            IsHTML:  nodeParams["isHTML"].(bool),
        })
        if err != nil {
            return nil, n.CreateError("failed to send email", map[string]interface{}{
                "to":    to,
                "error": err.Error(),
            })
        }

        // Create result item
        resultItem := model.DataItem{
            JSON: map[string]interface{}{
                "messageId": messageID,
                "to":        to,
                "subject":   subject,
                "sentAt":    time.Now().UTC().Format(time.RFC3339),
                "status":    "sent",
            },
        }

        results = append(results, resultItem)
    }

    return results, nil
}

type SMTPConfig struct {
    Host     string `json:"host"`
    Port     int    `json:"port"`
    Username string `json:"username"`
    Password string `json:"password"`
    TLS      bool   `json:"tls"`
}

type EmailMessage struct {
    To      string `json:"to"`
    Subject string `json:"subject"`
    Body    string `json:"body"`
    IsHTML  bool   `json:"isHTML"`
}

func (n *SMTPNode) sendEmail(config SMTPConfig, message EmailMessage) (string, error) {
    // Create SMTP client
    client, err := smtp.Dial(fmt.Sprintf("%s:%d", config.Host, config.Port))
    if err != nil {
        return "", err
    }
    defer client.Close()

    // Start TLS if required
    if config.TLS {
        tlsConfig := &tls.Config{ServerName: config.Host}
        if err = client.StartTLS(tlsConfig); err != nil {
            return "", err
        }
    }

    // Authenticate
    auth := smtp.PlainAuth("", config.Username, config.Password, config.Host)
    if err = client.Auth(auth); err != nil {
        return "", err
    }

    // Set sender and recipient
    if err = client.Mail(config.Username); err != nil {
        return "", err
    }
    if err = client.Rcpt(message.To); err != nil {
        return "", err
    }

    // Send message
    writer, err := client.Data()
    if err != nil {
        return "", err
    }

    // Generate message ID
    messageID := fmt.Sprintf("<%s@%s>", generateUUID(), config.Host)

    // Write headers and body
    fmt.Fprintf(writer, "To: %s\r\n", message.To)
    fmt.Fprintf(writer, "Subject: %s\r\n", message.Subject)
    fmt.Fprintf(writer, "Message-ID: %s\r\n", messageID)
    if message.IsHTML {
        fmt.Fprintf(writer, "Content-Type: text/html; charset=UTF-8\r\n")
    }
    fmt.Fprintf(writer, "\r\n%s\r\n", message.Body)

    if err = writer.Close(); err != nil {
        return "", err
    }

    return messageID, nil
}
```

#### SQL Database Node
```go
func (n *SQLDatabaseNode) Execute(inputData []model.DataItem, nodeParams map[string]interface{}) ([]model.DataItem, error) {
    var results []model.DataItem

    // Get database connection
    db, err := n.getDBConnection(nodeParams)
    if err != nil {
        return nil, n.CreateError("failed to connect to database", map[string]interface{}{
            "error": err.Error(),
        })
    }

    operation := nodeParams["operation"].(string)

    for index, item := range inputData {
        // Create expression context
        context := &expressions.ExpressionContext{
            ActiveNodeName:      "SQL Database",
            RunIndex:           0,
            ItemIndex:          index,
            Mode:               expressions.ModeManual,
            ConnectionInputData: []model.DataItem{item},
            AdditionalKeys:     make(map[string]interface{}),
        }

        switch operation {
        case "executeQuery":
            result, err := n.executeQuery(db, nodeParams, context)
            if err != nil {
                return nil, err
            }
            results = append(results, result)

        case "executeUpdate":
            result, err := n.executeUpdate(db, nodeParams, context)
            if err != nil {
                return nil, err
            }
            results = append(results, result)

        case "executeTransaction":
            result, err := n.executeTransaction(db, nodeParams, context)
            if err != nil {
                return nil, err
            }
            results = append(results, result)
        }
    }

    return results, nil
}

func (n *SQLDatabaseNode) executeQuery(db *sql.DB, nodeParams map[string]interface{}, context *expressions.ExpressionContext) (model.DataItem, error) {
    // Evaluate SQL query
    query, err := n.evaluator.EvaluateExpression(nodeParams["query"].(string), context)
    if err != nil {
        return model.DataItem{}, n.CreateError("failed to evaluate query", map[string]interface{}{
            "expression": nodeParams["query"],
            "error":      err.Error(),
        })
    }

    // Execute query
    rows, err := db.Query(query.(string))
    if err != nil {
        return model.DataItem{}, n.CreateError("failed to execute query", map[string]interface{}{
            "query": query,
            "error": err.Error(),
        })
    }
    defer rows.Close()

    // Get column names
    columns, err := rows.Columns()
    if err != nil {
        return model.DataItem{}, n.CreateError("failed to get columns", map[string]interface{}{
            "error": err.Error(),
        })
    }

    // Process results
    var results []map[string]interface{}
    for rows.Next() {
        // Create slice for column values
        values := make([]interface{}, len(columns))
        valuePtrs := make([]interface{}, len(columns))
        for i := range values {
            valuePtrs[i] = &values[i]
        }

        // Scan row
        if err := rows.Scan(valuePtrs...); err != nil {
            return model.DataItem{}, n.CreateError("failed to scan row", map[string]interface{}{
                "error": err.Error(),
            })
        }

        // Create result map
        rowMap := make(map[string]interface{})
        for i, col := range columns {
            rowMap[col] = values[i]
        }
        results = append(results, rowMap)
    }

    return model.DataItem{
        JSON: map[string]interface{}{
            "results":    results,
            "rowCount":   len(results),
            "columns":    columns,
            "query":      query,
            "executedAt": time.Now().UTC().Format(time.RFC3339),
        },
    }, nil
}
```

#### Workflow Scheduler
```go
func (s *WorkflowScheduler) AddSchedule(config ScheduleConfig) error {
    s.mutex.Lock()
    defer s.mutex.Unlock()

    // Validate cron expression
    if _, err := cron.ParseStandard(config.CronExpr); err != nil {
        return fmt.Errorf("invalid cron expression: %w", err)
    }

    // Create new cron job
    c := cron.New(cron.WithLocation(s.getTimezone(config.Timezone)))

    _, err := c.AddFunc(config.CronExpr, func() {
        s.executeScheduledWorkflow(config.WorkflowID)
    })
    if err != nil {
        return fmt.Errorf("failed to add cron job: %w", err)
    }

    // Start the cron job
    if config.Enabled {
        c.Start()
    }

    // Store the schedule
    scheduleID := generateScheduleID(config.WorkflowID, config.CronExpr)
    s.cronJobs[scheduleID] = c
    s.schedules[scheduleID] = &config

    return nil
}

func (s *WorkflowScheduler) executeScheduledWorkflow(workflowID string) {
    // Load workflow
    workflow, err := s.loadWorkflow(workflowID)
    if err != nil {
        log.Printf("Failed to load workflow %s: %v", workflowID, err)
        return
    }

    // Execute workflow
    result, err := s.engine.ExecuteWorkflow(workflow, []model.DataItem{})
    if err != nil {
        log.Printf("Scheduled execution failed for workflow %s: %v", workflowID, err)
        return
    }

    // Log successful execution
    log.Printf("Scheduled workflow %s executed successfully. Duration: %v",
        workflowID, result.ExecutionTime)

    // Update schedule statistics
    s.updateScheduleStats(workflowID, err == nil)
}
```

```go
    status := "success"
    if !success {
        status = "failure"

    m.nodeExecutions.WithLabelValues(nodeType, status).Inc()
    m.executionDuration.WithLabelValues(nodeType).Observe(duration.Seconds())
```
